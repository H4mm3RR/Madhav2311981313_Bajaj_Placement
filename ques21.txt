class Solution {
    public int[] searchRange(int[] nums, int target) {
        int f=firstocc(nums,target);
        int last=lastocc(nums,target);
        if(f!=-1&&last!=-1){
            return new int[]{f,last};
        }
        return new int[]{-1,-1};
    }
    public int firstocc(int[] nums,int target){
        int n=nums.length;
        int l=0;
        int h=n-1;
        int idx1=-1;
        while(l<=h){
            int m=l+(h-l)/2;
            if(nums[m]==target){
                idx1=m;
                h=m-1;
            }else if(nums[m]<target){
                l=m+1;
            }else{
                h=m-1;
            }
        }
        return idx1;
    }
    public int lastocc(int[] nums,int target){
        int n=nums.length;
        int l=0;
        int h=n-1;
        int idx2=-1;
        while(l<=h){
            int m=l+(h-l)/2;
            if(nums[m]==target){
                idx2=m;
                l=m+1;
            }else if(nums[m]<target){
                l=m+1;
            }else{
                h=m-1;
            }
        }
        return idx2;
    }
}
